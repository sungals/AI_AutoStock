# 백테스트 신뢰성 레이어 — 구현 플랜

> **For agentic workers (Claude/Codex 공통):** 이 플랜은 task 단위로 TDD(실패테스트 → 최소구현 → 통과 → 커밋)로 구현한다.
> 각 task는 독립적으로 검증 가능한 산출물로 끝난다. cold-start 시 `00-스펙-설계.md`와
> `codex-handoff/00-codex-시작가이드.md`를 먼저 읽을 것.

**Goal:** 기존 TTAK Quant 백테스트/시뮬레이션 엔진에 미래참조 차단·체결현실성·과최적화 방지·생존편향 경고를 점진적으로 추가한다.

**Architecture:** Business Layer에 순수함수 모듈 3개(`point_in_time`, `execution_model`, `validation_harness`) + 리포트 1개(`bias_report`)를 추가하고, `backtester`/`simulation_runner`/`algo_optimizer`가 이를 경유하게 최소 수정한다.

**Tech Stack:** Python 3.9 (타입힌트는 `Optional[X]` 사용, `X|None` 금지), SQLite3(WAL), pytest, numpy.

## Global Constraints (모든 task에 적용)

- Python 3.9 호환 타입힌트만 사용 (`from typing import Optional, List, Dict, Tuple`).
- DB 함수는 `conn`을 주입받는다 (Connection Injection). 순수함수 모듈은 DB 접근 금지.
- 마이그레이션은 `PRAGMA table_info`로 컬럼 존재 확인 후 추가 (멱등).
- SQL은 파라미터 바인딩만 (f-string 쿼리 금지).
- 모든 신규 모듈 파일은 단일 책임. 테스트는 `tests/`에 위치.
- 설정값은 `config.py`에서 읽는다 (하드코딩 금지).

```python
# point_in_time.py
from typing import Optional, List, Dict
from datetime import date, timedelta
import config

def estimate_disclosed_at(bsns_year: str, reprt_code: str) -> str:
    year = int(bsns_year)
    if reprt_code == '11011':       # 연간
        base = date(year, 12, 31); lag = config.PIT_ANNUAL_LAG_DAYS
    elif reprt_code == '11013':     # 1Q
        base = date(year, 3, 31);  lag = config.PIT_QUARTER_LAG_DAYS
    elif reprt_code == '11012':     # 반기
        base = date(year, 6, 30);  lag = config.PIT_QUARTER_LAG_DAYS
    elif reprt_code == '11014':     # 3Q
        base = date(year, 9, 30);  lag = config.PIT_QUARTER_LAG_DAYS
    else:
        base = date(year, 12, 31);  lag = config.PIT_ANNUAL_LAG_DAYS
    return (base + timedelta(days=lag)).isoformat()

def get_financials_asof(conn, corp_code: str, as_of_date: str) -> List[Dict]:
    rows = conn.execute(
        "SELECT * FROM financial_statements WHERE corp_code=?", (corp_code,)
    ).fetchall()
    out = []
    for r in rows:
        d = dict(r)
        disclosed = d.get('disclosed_at') or estimate_disclosed_at(
            d['bsns_year'], d['reprt_code'])
        if disclosed <= as_of_date:
            d['disclosed_at'] = disclosed
            out.append(d)
    return out

def get_metrics_asof(conn, corp_code: str, as_of_date: str) -> Dict:
    # PIT 재무 + as_of 시세로 재계산 (metrics_calculator 재사용)
    import metrics_calculator as mc
    fins = get_financials_asof(conn, corp_code, as_of_date)
    # ... fins에서 필요한 계정 추출 후 mc.calc_* 호출 (스펙 §4.1)
    return {}  # 실제 계산은 metrics_calculator 연동 시 채움
```

```python
# execution_model.py
from typing import Dict
import config

def sell_tax_rate(trade_date: str) -> float:
    for eff_date, rate in config.SELL_TAX_TABLE:   # 최신순 정렬 가정
        if trade_date >= eff_date:
            return rate
    return config.SELL_TAX_TABLE[-1][1]

def apply_costs(side: str, price: float, qty: int, trade_date: str,
                commission_rate: float, slippage_bps: float) -> Dict:
    slip = price * (slippage_bps / 10000.0)
    fill_price = price + slip if side == 'buy' else price - slip
    gross = fill_price * qty
    commission = gross * commission_rate
    tax = gross * sell_tax_rate(trade_date) if side == 'sell' else 0.0
    if side == 'buy':
        net = gross + commission           # 실지불
    else:
        net = gross - commission - tax      # 실수령
    return {'fill_price': fill_price, 'gross': gross, 'commission': commission,
            'tax': tax, 'slippage': slip * qty, 'net': net}
```

```python
def round_to_tick(price: float) -> int:
    p = float(price)
    for upper, tick in config.TICK_TABLE:
        if p < upper:
            return int(round(p / tick) * tick)
    return int(round(p / 1000) * 1000)

def is_tradable(side: str, ohlcv: Dict, prev_close: int,
                min_value_krw: int = None) -> bool:
    if min_value_krw is None:
        min_value_krw = config.MIN_TRADE_VALUE_KRW
    if not ohlcv or ohlcv.get('volume', 0) <= 0:
        return False
    high, low, close = ohlcv.get('high', 0), ohlcv.get('low', 0), ohlcv.get('close', 0)
    limit_up = prev_close * (1 + config.PRICE_LIMIT_PCT)
    limit_down = prev_close * (1 - config.PRICE_LIMIT_PCT)
    locked = (high == low)
    if side == 'buy' and locked and close >= limit_up * 0.999:
        return False
    if side == 'sell' and locked and close <= limit_down * 1.001:
        return False
    value = ohlcv.get('value') or (close * ohlcv.get('volume', 0))
    if value < min_value_krw:
        return False
    return True
```
