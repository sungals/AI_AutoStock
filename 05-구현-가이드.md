```python
# auth.py
from werkzeug.security import generate_password_hash, check_password_hash
from flask_login import UserMixin
import db

class User(UserMixin):
    def __init__(self, id, username, is_admin=False):
        self.id = id
        self.username = username
        self.is_admin = is_admin

def create_user(username: str, password: str, is_admin: bool = False):
    """새 사용자 생성. 첫 번째 사용자를 관리자로 설정."""
    password_hash = generate_password_hash(password)
    with db.get_connection() as conn:
        conn.execute(
            "INSERT INTO users (username, password_hash, is_admin) VALUES (?, ?, ?)",
            (username, password_hash, 1 if is_admin else 0)
        )

def verify_user(username: str, password: str):
    """사용자명/비밀번호 검증. 성공 시 User 객체 반환."""
    with db.get_connection() as conn:
        row = conn.execute(
            "SELECT * FROM users WHERE username=?", (username,)
        ).fetchone()
    if row and check_password_hash(row['password_hash'], password):
        return User(row['id'], row['username'], bool(row['is_admin']))
    return None
```

### 2.1 기업 마스터 (company_master.py)

DART의 기업 고유번호와 KRX의 종목 코드를 매핑합니다.

```python
# company_master.py
import requests
import zipfile
import io
import xml.etree.ElementTree as ET
from pykrx import stock
import db

DART_CORPCODE_URL = "https://opendart.fss.or.kr/api/corpCode.xml"

def _fetch_dart_corp_codes(api_key: str) -> dict:
    """DART에서 전체 기업 고유번호 목록 다운로드.

    Returns:
        dict: {stock_code: {corp_code, corp_name}}
    """
    params = {"crtfc_key": api_key}
    response = requests.get(DART_CORPCODE_URL, params=params)

    with zipfile.ZipFile(io.BytesIO(response.content)) as z:
        xml_content = z.read('CORPCODE.xml')

    root = ET.fromstring(xml_content)
    result = {}
    for item in root.findall('list'):
        stock_code = item.findtext('stock_code', '').strip()
        if stock_code:  # 상장 기업만 (미상장은 stock_code 없음)
            result[stock_code] = {
                'corp_code': item.findtext('corp_code', '').strip(),
                'corp_name': item.findtext('corp_name', '').strip(),
            }
    return result

def build_company_master() -> list:
    """DART + pykrx 데이터를 매핑하여 companies 테이블 구축."""
    import os
    api_key = os.environ.get('OPENDART_API_KEY', '')

    # DART corp_code 목록
    dart_companies = _fetch_dart_corp_codes(api_key)

    # pykrx KOSPI 종목
    kospi_tickers = stock.get_market_ticker_list(market='KOSPI')
    kosdaq_tickers = stock.get_market_ticker_list(market='KOSDAQ')

    companies = []
    with db.get_connection() as conn:
        for market, tickers in [('KOSPI', kospi_tickers), ('KOSDAQ', kosdaq_tickers)]:
            for ticker in tickers:
                dart_info = dart_companies.get(ticker)
                if not dart_info:
                    continue

                corp_name = stock.get_market_ticker_name(ticker) or dart_info['corp_name']

                conn.execute("""
                    INSERT OR REPLACE INTO companies
                        (corp_code, stock_code, corp_name, market)
                    VALUES (?, ?, ?, ?)
                """, (dart_info['corp_code'], ticker, corp_name, market))
                companies.append((dart_info['corp_code'], ticker, corp_name))

    return companies
```

### 2.3 재무 지표 계산 (metrics_calculator.py)

```python
# metrics_calculator.py

def calc_per(market_cap: int, net_income: int) -> Optional[float]:
    """PER = 시가총액 / 당기순이익"""
    if not market_cap or not net_income or net_income <= 0:
        return None
    return round(market_cap / net_income, 2)

def calc_pbr(market_cap: int, equity: int) -> Optional[float]:
    """PBR = 시가총액 / 자본총계"""
    if not market_cap or not equity or equity <= 0:
        return None
    return round(market_cap / equity, 2)

def calc_roe(net_income: int, equity: int) -> Optional[float]:
    """ROE = 당기순이익 / 자본총계 × 100"""
    if not net_income or not equity or equity <= 0:
        return None
    return round(net_income / equity * 100, 2)

def calc_debt_ratio(liabilities: int, equity: int) -> Optional[float]:
    """부채비율 = 부채총계 / 자본총계 × 100"""
    if liabilities is None or not equity or equity <= 0:
        return None
    return round(liabilities / equity * 100, 2)

def calc_revenue_growth(current: int, previous: int) -> Optional[float]:
    """매출 성장률 (전년 대비 %)"""
    if not previous or previous <= 0 or current is None:
        return None
    return round((current - previous) / previous * 100, 2)
```

```python
import pandas as pd
import ta

def calculate_technical_signals(stock_code: str, df: pd.DataFrame) -> dict:
    """df: OHLCV DataFrame (ascending order)
    Returns: {signal_name: (value, label)}
    """
    close = df['close']
    high = df['high']
    low = df['low']
    volume = df['volume']
    results = {}

    # --- 추세 지표 ---
    # MA
    for period in [5, 20, 60, 120, 200]:
        if len(df) >= period:
            ma = close.rolling(period).mean()
            val = float(ma.iloc[-1]) if pd.notna(ma.iloc[-1]) else None
            results[f'ma_{period}'] = (val, f'MA{period}={val:,.0f}' if val else None)

    # 골든크로스 / 데드크로스
    if len(df) >= 21:
        ma5 = close.rolling(5).mean()
        ma20 = close.rolling(20).mean()
        curr_5, curr_20 = float(ma5.iloc[-1]), float(ma20.iloc[-1])
        prev_5, prev_20 = float(ma5.iloc[-2]), float(ma20.iloc[-2])
        if curr_5 > curr_20 and prev_5 <= prev_20:
            cross = (1, '골든크로스')
        elif curr_5 < curr_20 and prev_5 >= prev_20:
            cross = (-1, '데드크로스')
        else:
            cross = (0, '보합')
        results['ma_cross'] = cross

    # MACD
    if len(df) >= 35:
        macd_ind = ta.trend.MACD(close)
        macd_val = float(macd_ind.macd().iloc[-1])
        macd_sig = float(macd_ind.macd_signal().iloc[-1])
        macd_hist = float(macd_ind.macd_diff().iloc[-1])
        label = 'MACD 골든크로스' if macd_hist > 0 else 'MACD 데드크로스'
        results['macd'] = (macd_val, label)

    # --- 모멘텀 지표 ---
    # RSI-14
    if len(df) >= 15:
        rsi = ta.momentum.RSIIndicator(close, window=14).rsi()
        val = float(rsi.iloc[-1])
        if val <= 30:
            label = f'RSI 과매도 ({val:.0f})'
        elif val >= 70:
            label = f'RSI 과매수 ({val:.0f})'
        else:
            label = f'RSI 중립 ({val:.0f})'
        results['rsi_14'] = (val, label)

    # Bollinger Bands
    if len(df) >= 20:
        bb = ta.volatility.BollingerBands(close, window=20, window_dev=2)
        curr_close = float(close.iloc[-1])
        bb_lower = float(bb.bollinger_lband().iloc[-1])
        bb_upper = float(bb.bollinger_hband().iloc[-1])
        if curr_close <= bb_lower:
            label = '볼린저 하단 이탈 (과매도)'
        elif curr_close >= bb_upper:
            label = '볼린저 상단 이탈 (과매수)'
        else:
            label = '볼린저 밴드 내'
        results['bollinger'] = (curr_close, label)

    # ATR-14
    if len(df) >= 15:
        atr = ta.volatility.AverageTrueRange(high, low, close, window=14).average_true_range()
        val = float(atr.iloc[-1])
        results['atr_14'] = (val, f'ATR={val:,.0f}')

    # OBV
    if len(df) >= 5:
        obv = ta.volume.OnBalanceVolumeIndicator(close, volume).on_balance_volume()
        val = float(obv.iloc[-1])
        results['obv'] = (val, f'OBV={val:,.0f}')

    return results
```

### 3.2 캔들스틱 패턴 감지

```python
def detect_candlestick_patterns(df: pd.DataFrame) -> dict:
    """4종 캔들스틱 패턴 감지."""
    if len(df) < 2:
        return {}

    results = {}
    r = df.iloc[-1]  # 오늘
    p = df.iloc[-2]  # 어제

    body = abs(r['close'] - r['open'])
    upper_shadow = r['high'] - max(r['close'], r['open'])
    lower_shadow = min(r['close'], r['open']) - r['low']

    # 망치형 (Hammer): 하단 꼬리 > 몸통 × 2, 양봉
    if lower_shadow > body * 2 and upper_shadow < body * 0.5 and r['close'] > r['open']:
        results['hammer'] = (1, '망치형 (반등 신호)')

    # 도지 (Doji): 시가 ≈ 종가
    atr_approx = (df['high'] - df['low']).rolling(14).mean().iloc[-1]
    if body < atr_approx * 0.1:
        results['doji'] = (1, '도지 (방향 전환 신호)')

    # 상승장악형 (Bullish Engulfing): 전일 음봉 < 당일 양봉
    prev_body = abs(p['close'] - p['open'])
    if (p['close'] < p['open'] and r['close'] > r['open'] and
            r['open'] < p['close'] and r['close'] > p['open']):
        results['bullish_engulfing'] = (1, '상승장악형 (강한 반등 신호)')

    # 유성형 (Shooting Star): 상단 꼬리 > 몸통 × 2, 음봉
    if upper_shadow > body * 2 and lower_shadow < body * 0.5 and r['close'] < r['open']:
        results['shooting_star'] = (-1, '유성형 (하락 신호)')

    return results
```

```python
POSITIVE_KEYWORDS = [
    '급등', '상승', '호실적', '흑자', '성장', '최고', '돌파', '신고가',
    '매수', '긍정', '개선', '확대', '증가', '호조', '선방', '수주',
    '계약', '특허', '혁신', '구조조정', '흑자전환', '실적개선',
]

NEGATIVE_KEYWORDS = [
    '급락', '하락', '부진', '적자', '감소', '최저', '이탈', '신저가',
    '매도', '부정', '악화', '축소', '감소', '우려', '손실', '구조조정',
    '소송', '벌금', '횡령', '수사', '적자전환', '실적악화',
]

def analyze_sentiment(text: str) -> tuple:
    """텍스트의 감성을 분석합니다.

    Returns:
        (sentiment, score): ('positive'/'negative'/'neutral', -1.0 ~ 1.0)
    """
    pos_count = sum(1 for kw in POSITIVE_KEYWORDS if kw in text)
    neg_count = sum(1 for kw in NEGATIVE_KEYWORDS if kw in text)

    total = pos_count + neg_count
    if total == 0:
        return 'neutral', 0.0

    score = (pos_count - neg_count) / total

    if score > 0.1:
        return 'positive', round(score, 3)
    elif score < -0.1:
        return 'negative', round(score, 3)
    else:
        return 'neutral', round(score, 3)
```

```python
    recent_close = closes[0]
    close_60d = closes[-1]

    else:
        return 'sideways'
```

```python
# fusion_analyzer.py

def calculate_fusion_signal(
    tech_score: float,    # 기술적 분석 점수 (-100~100)
    emp_score: float,     # 경험적 분석 점수 (-100~100)
    tech_weight: float = 0.5,
    emp_weight: float = 0.5,
    amplifier: float = 1.2,   # 동의 시 증폭 계수
    dampener: float = 0.7,    # 불일치 시 감소 계수
) -> dict:
    """기술적 + 경험적 점수를 융합하여 최종 시그널 계산."""

    # 기본 융합 점수
    fusion_score = tech_score * tech_weight + emp_score * emp_weight

    # 동의/불일치 판단
    agreement = (tech_score > 0 and emp_score > 0) or (tech_score < 0 and emp_score < 0)

    if agreement:
        fusion_score *= amplifier  # 동의 증폭
        confidence = 0.8
    else:
        fusion_score *= dampener   # 불일치 완화
        confidence = 0.4

    # 추천 결정
    if fusion_score >= 60:
        recommendation = 'STRONG_BUY'
    elif fusion_score >= 30:
        recommendation = 'BUY'
    elif fusion_score <= -60:
        recommendation = 'STRONG_SELL'
    elif fusion_score <= -30:
        recommendation = 'SELL'
    else:
        recommendation = 'HOLD'

    return {
        'fusion_score': round(fusion_score, 2),
        'confidence': confidence,
        'agreement': 1 if agreement else 0,
        'recommendation': recommendation,
    }
```

```python
# simulation_runner.py 핵심 구조
from concurrent.futures import ThreadPoolExecutor
import itertools
import uuid
import db

def run_batch(n: int = 10000) -> dict:
    """n회 walk-forward 시뮬레이션 병렬 실행."""
    batch_id = str(uuid.uuid4())

    # 파라미터 공간
    strategies = ['turnaround', 'value', 'momentum', 'flow', 'guru']
    risk_profiles = ['aggressive', 'moderate', 'defensive']
    window_months = [3, 6, 12]
    start_offsets = range(0, 36, 3)   # 0~33개월 오프셋

    params = list(itertools.product(strategies, risk_profiles, window_months, start_offsets))

    # n개로 제한 (순환 사용)
    param_list = [params[i % len(params)] for i in range(n)]

    results = {'completed': 0, 'failed': 0}

    with ThreadPoolExecutor(max_workers=4) as executor:
        futures = [
            executor.submit(_run_single_simulation, batch_id, *p)
            for p in param_list
        ]
        for future in futures:
            try:
                future.result()
                results['completed'] += 1
            except Exception:
                results['failed'] += 1

    return results

def _run_single_simulation(batch_id, strategy, risk_profile, window_months, start_offset):
    """단일 시뮬레이션 실행."""
    from backtester import run_backtest
    from datetime import date, timedelta

    # 날짜 계산
    today = date.today()
    end = today - timedelta(days=start_offset * 30)
    start = end - timedelta(days=window_months * 30)

    result = run_backtest(
        strategy=strategy,
        start_date=start.strftime('%Y-%m-%d'),
        end_date=end.strftime('%Y-%m-%d'),
        risk_profile=risk_profile,
    )

    with db.get_connection() as conn:
        conn.execute("""
            INSERT INTO simulation_runs
                (batch_id, strategy, risk_profile, window_months, start_offset_months,
                 cagr, mdd, sharpe, alpha, n_trades, market_regime, status)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 'ok')
        """, (batch_id, strategy, risk_profile, window_months, start_offset,
              result.get('cagr'), result.get('mdd'), result.get('sharpe'),
              result.get('alpha'), result.get('n_trades'), result.get('regime')))
```

### 8.2 CAGR / MDD / Sharpe 계산

```python
# backtester.py

def calc_cagr(portfolio_values: list, years: float) -> float:
    """연평균 복리 수익률."""
    if years <= 0 or portfolio_values[0] <= 0:
        return 0.0
    total_return = portfolio_values[-1] / portfolio_values[0]
    return (total_return ** (1 / years) - 1) * 100

def calc_mdd(portfolio_values: list) -> float:
    """최대낙폭 (Maximum Drawdown, %)."""
    peak = portfolio_values[0]
    mdd = 0.0
    for v in portfolio_values:
        if v > peak:
            peak = v
        dd = (peak - v) / peak
        if dd > mdd:
            mdd = dd
    return mdd * 100

def calc_sharpe(daily_returns: list, risk_free_rate: float = 0.035) -> float:
    """Sharpe Ratio (연간화)."""
    import numpy as np
    if len(daily_returns) < 5:
        return 0.0
    arr = np.array(daily_returns)
    excess = arr - risk_free_rate / 252
    if arr.std() == 0:
        return 0.0
    return float(np.mean(excess) / arr.std() * (252 ** 0.5))
```

### 9.1 Kelly Criterion

```python
# kelly_criterion.py

def calculate_kelly_ratio(
    win_rate: float,      # 승률 (0~1)
    avg_win: float,       # 평균 수익률
    avg_loss: float,      # 평균 손실률 (양수)
    fraction: float = 0.25,  # 분수 Kelly (보수적 적용)
) -> float:
    """Kelly Criterion으로 최적 포지션 크기 계산.

    full_kelly = (bp - q) / b
    b = 평균 수익 / 평균 손실 (손익비)
    p = 승률, q = 1 - p
    """
    if avg_loss <= 0:
        return 0.0

    b = avg_win / avg_loss  # 손익비
    p = win_rate
    q = 1 - win_rate

    full_kelly = (b * p - q) / b

    # 분수 Kelly 적용 (과도한 배팅 방지)
    kelly = max(0.0, min(full_kelly * fraction, 0.3))  # 최대 30% 제한
    return round(kelly, 4)
```
